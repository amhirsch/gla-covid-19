sns.set()

def adjust_x_ax(observations):
    min_date = observations[const.DATE].min()
    max_date = observations[const.DATE].max()
    plt.xlim((min_date, max_date))
    plt.xticks(pd.date_range(min_date, max_date, freq='SMS'), rotation=45)

def adjust_y_ax(observations, dep_var):
    TOP_MARGIN = 1.1
    NUM_TICKS = 5
    max_value = math.ceil(observations[dep_var].max() * TOP_MARGIN)
    step_size = math.floor(max_value / NUM_TICKS)
    plt.ylim(0, max_value)
    plt.yticks(tuple(range(0, max_value, step_size)))
